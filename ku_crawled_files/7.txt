            if (navigator.appVersion.match(/MSIE/) && window.XDomainRequest) {
                // Use Microsoft XDR
                var xdr = new XDomainRequest();
                xdr.open("get", url);
                xdr.onload = function () {
                    var result = weatherval+xdr.responseText;
                    $kuj.cookie('myweather', result, { expires: date });
                    $kuj('#weatherholder').html(result);
                };
                xdr.send();
            } else if (!navigator.appVersion.match(/MSIE/)) {
                $kuj.ajax({
                    type: 'GET',
                    url: url,
                    success: function (data) {
                        date.setTime(date.getTime() + (minutes * 60 * 1000));
                        $kuj.cookie('myweather', weatherval+data, { expires: date });
                        $kuj('#weatherholder').html($kuj.cookie('myweather'));
                    }
                });
            }
        } catch (e) {}
        }else{
            $kuj('#weatherholder').html($kuj.cookie('myweather'));
        }

    });
</script>

        <div class="field field-name-body field-type-text-with-summary field-label-hidden"><div class="field-items"><div class="field-item even" property="content:encoded"><h1>Office of the General Counsel
</h1>
<p><strong>Mission</strong>:&nbsp;Provide legal advice, counsel, and representation to the University of Kansas through the Chancellor and those she designates.</p>

<p><strong>Vision</strong>:&nbsp;A professional staff with core competencies of unquestioned commitment to ethical behavior, thorough and effective research, accurate legal advice, effective preventive law, and successful representation of KU's interests in all legal and administrative settings.</p>

<h2>A "One-University" office with legal support to:</h2>

<ul>
	<li><a href="http://www.ku.edu/">Lawrence campus</a></li>
	<li>Medical Center campuses in&nbsp;<a href="http://www.kumc.edu/">Kansas City</a>,&nbsp;<a href="http://wichita.kumc.edu/">Wichita</a>, and&nbsp;<a href="http://salina.kumc.edu/">Salina</a></li>
	<li><a href="http://www.edwardscampus.ku.edu/">Edwards Campus</a></li>
	<li><a href="http://www.kletc.org/">Kansas Law Enforcement Training Center</a>&nbsp;in Hutchinson</li>
</ul>

<p>The Office of the General Counsel is responsible for legal matters of the University of Kansas. In carrying out that responsibility, the General Counsel and other members of the office draft legal documents, review and recommend action on University contracts and agreements, represent the University in legal actions, and provide legal advice to those who administer the affairs of the University. For the purposes of prosecuting or defending actions involving the University, the attorneys in the Office of the General Counsel serve as Special Assistant Attorneys General by appointment of the Attorney General of the State of Kansas.</p>
