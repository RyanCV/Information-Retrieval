    //Use javascript to avoid weather caching
    $kuj(document).ready(function() {
        if(!$kuj.cookie('myweather')){
            var url = '//webmedia.ku.edu/~http/ssi/2012/widgets/cmslawrenceweather.php';
            var weatherval = '<div class="sectiontitle">Current Weather</div>';
            var date = new Date();
            var minutes = 30;
            date.setTime(date.getTime() + (minutes * 60 * 1000));
            try {
            //cross script issue with IE8 workaround
            if (navigator.appVersion.match(/MSIE/) && window.XDomainRequest) {
                // Use Microsoft XDR
                var xdr = new XDomainRequest();
                xdr.open("get", url);
                xdr.onload = function () {
                    var result = weatherval+xdr.responseText;
                    $kuj.cookie('myweather', result, { expires: date });
                    $kuj('#weatherholder').html(result);
                };
                xdr.send();
            } else if (!navigator.appVersion.match(/MSIE/)) {
                $kuj.ajax({
                    type: 'GET',
                    url: url,
                    success: function (data) {
                        date.setTime(date.getTime() + (minutes * 60 * 1000));
                        $kuj.cookie('myweather', weatherval+data, { expires: date });
                        $kuj('#weatherholder').html($kuj.cookie('myweather'));
                    }
                });
            }
        } catch (e) {}
        }else{
            $kuj('#weatherholder').html($kuj.cookie('myweather'));
        }

    });
</script>

                                <h1 class="title" id="page-title">Responsible Scholarship</h1>
                    <span property="dc:title" content="Responsible Scholarship" class="rdf-meta element-hidden"></span>    
    <div class="content clearfix">
        <div class="field field-name-body field-type-text-with-summary field-label-hidden"><div class="field-items"><div class="field-item even" property="content:encoded"><div>
<p>Responsible Scholarship is essential for all involved with the discovery, application and dissemination of knowledge. Responsible Scholarship encompasses professional behaviors and practices in the design, implementation and reporting of academic scholarship. The following list contains links to online tutorials, recommended textbooks, and audio-visual materials available for checkout from Research Integrity.  DVD, CD or VHS of case studies available from Research Integrity, please contact Susan MacNally for more information.</p>
