    //Use javascript to avoid weather caching
    $kuj(document).ready(function() {
        if(!$kuj.cookie('myweather')){
            var url = '//webmedia.ku.edu/~http/ssi/2012/widgets/cmslawrenceweather.php';
            var weatherval = '<div class="sectiontitle">Current Weather</div>';
            var date = new Date();
            var minutes = 30;
            date.setTime(date.getTime() + (minutes * 60 * 1000));
            try {
            //cross script issue with IE8 workaround
            if (navigator.appVersion.match(/MSIE/) && window.XDomainRequest) {
                // Use Microsoft XDR
                var xdr = new XDomainRequest();
                xdr.open("get", url);
                xdr.onload = function () {
                    var result = weatherval+xdr.responseText;
                    $kuj.cookie('myweather', result, { expires: date });
                    $kuj('#weatherholder').html(result);
                };
                xdr.send();
            } else if (!navigator.appVersion.match(/MSIE/)) {
                $kuj.ajax({
                    type: 'GET',
                    url: url,
                    success: function (data) {
                        date.setTime(date.getTime() + (minutes * 60 * 1000));
                        $kuj.cookie('myweather', weatherval+data, { expires: date });
                        $kuj('#weatherholder').html($kuj.cookie('myweather'));
                    }
                });
            }
        } catch (e) {}
        }else{
            $kuj('#weatherholder').html($kuj.cookie('myweather'));
        }

    });

<p>Located on the third floor of Chalmers Hall, the Art &amp; Design Gallery serves as&nbsp;a teaching space. Students from the department, as well as from the university at large, are exposed to art by their peers, visiting and regional artists, and faculty members.&nbsp;Many Visual Art students experience installing and exhibiting their work in a gallery setting for the first time in this space.&nbsp;The 2,100 square foot gallery has been recently renovated. The gallery is free and open to the public.</p>
