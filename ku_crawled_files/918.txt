        /// Shows a schedule for a given group id
        function showSchedule(groupId) {
            $("#dgSchedule").dialog("open");

            $('#calendar').fullCalendar("destroy");

            $('#calendar').fullCalendar($.extend({
                header: {
                    left: 'prev,next today',
                    center: 'title',
                    right: 'month,agendaWeek,agendaDay'
                },
                defaultView: 'agendaWeek',
                editable: false,
                disableDragging: false,
                disableResizing: false,
                events: function (start, end, callback) {
                    $.ajax({
                        type: "POST",
                        url: "/LabStats/GroupsStations/HierarchyEditor/he_groupSchedule.aspx/getEvents",
                        dataType: 'json',
                        contentType: "application/json; charset=utf-8",
                        ignoreTimezone: false,
                        data: JSON.stringify({
                            groupId: groupId,
                            start: start,
                            end: end,
                            isPublic: true
                        }),
                        success: function (doc) {

                            var events = [];

                            $(doc.d).each(function (key, value) {
                                events.push({
                                    id: value['id'],
                                    title: value['title'],
                                    start: value['startAsIsoString'],
                                    end: value['endAsIsoString'],
                                    className: "eventColor" + value['color'],
                                    color: value['color'],
                                    recurrenceParentId: value['recurrenceParentID'],
                                    isExpandedEvent: value['isExpandedEvent'],
                                    frequency: value['frequency'],
                                    interval: value['interval'],
                                    recursUntil: value['recursUntilAsIsoString'],
                                    recurCount: value['recurCount'],
                                    recurrenceStart: value['recurrenceStart'],
                                    eventDuration: value['eventDuration'],
                                    groupId: value['groupId'],
                                    status: value['status'],
                                    publishLevel: value['publishLevel'],
                                    daysOfWeekMask: value['daysOfWeekMask'],
                                    dayOfMonth: value['dayOfMonth'],
                                    dayOrdinal: value['dayOrdinal'],
                                    recurrenceMonth: value['recurrenceMonth'],
                                    allDay: false,
                                    ignoreTimezone: false,
                                    editable: false
                                });
                            });

                            callback(events);
                        }
                    });
                },
                theme: true,
                eventClick: function (event, jsEvent, view) {
                    showInfoDialog(event);
                }
            }, fullCalendarLocalization));
        }

        function showInfoDialog(event) {
            var dateConverter = new AnyTime.Converter({ format: "%m/%d/%Y %h:%i %p" });

            $("#eventTitleInfo").html(event.title);
            $("#eventStartTimeInfo").html(dateConverter.format(event.start));
            $("#eventEndTimeInfo").html(dateConverter.format(event.end));

            $("#infoDialog").dialog({ width: '420px', title: 'Event Details' });
        }

        // A utility for dumping an object's contents
        function dump(arr, level) {
            var dumped_text = "";
            if (!level) level = 0;

            //The padding given at the beginning of the line.
            var level_padding = "";
            for (var j = 0; j < level + 1; j++) level_padding += "    ";

            if (typeof (arr) == 'object') { //Array/Hashes/Objects 
                for (var item in arr) {
                    var value = arr[item];

                    if (typeof (value) == 'object') { //If it is an array,
                        dumped_text += level_padding + "'" + item + "' ...\n";
                        dumped_text += dump(value, level + 1);
                    } else {
                        dumped_text += level_padding + "'" + item + "' => \"" + value + "\"\n";
                    }
                }
            } else { //Stings/Chars/Numbers etc.
                dumped_text = "===>" + arr + "<===(" + typeof (arr) + ")";
            }
            return dumped_text;
        }
