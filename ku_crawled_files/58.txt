    //Use javascript to avoid weather caching
    $kuj(document).ready(function() {
        if(!$kuj.cookie('myweather')){
            var url = '//webmedia.ku.edu/~http/ssi/2012/widgets/cmslawrenceweather.php';
            var weatherval = '<div class="sectiontitle">Current Weather</div>';
            var date = new Date();
            var minutes = 30;
            date.setTime(date.getTime() + (minutes * 60 * 1000));
            try {
            //cross script issue with IE8 workaround
            if (navigator.appVersion.match(/MSIE/) && window.XDomainRequest) {
                // Use Microsoft XDR
                var xdr = new XDomainRequest();
                xdr.open("get", url);
                xdr.onload = function () {
                    var result = weatherval+xdr.responseText;
                    $kuj.cookie('myweather', result, { expires: date });
                    $kuj('#weatherholder').html(result);
                };
                xdr.send();
            } else if (!navigator.appVersion.match(/MSIE/)) {
                $kuj.ajax({
                    type: 'GET',
                    url: url,
                    success: function (data) {
                        date.setTime(date.getTime() + (minutes * 60 * 1000));
                        $kuj.cookie('myweather', weatherval+data, { expires: date });
                        $kuj('#weatherholder').html($kuj.cookie('myweather'));
                    }
                });
            }
        } catch (e) {}
        }else{
            $kuj('#weatherholder').html($kuj.cookie('myweather'));
        }

    });

        <div class="field field-name-field-bottomleft field-type-text-long field-label-hidden"><div class="field-items"><div class="field-item even"><h2>Welcome to the Hall Center</h2>

<p style="margin-top: 0px; font-family: Arial, Helvetica, sans-serif; color: rgb(68, 68, 68); line-height: 17.71875px;">The Hall Center's primary mission is to stimulate and support research in the humanities, arts and social sciences, especially of an interdisciplinary kind, at the University of Kansas. The Center brings together faculty and graduate students with common interests from various disciplines to enable them to build on each others' ideas and to share their knowledge within the university and with the wider community.</p>

<p style="margin-top: 0px; font-family: Arial, Helvetica, sans-serif; color: rgb(68, 68, 68); line-height: 17.71875px;">The Center's collateral mission is to sponsor special programs that engage the university and the wider community in dialogue on issues that bring the humanities to bear on the quality of life for all citizens. It creates events on and beyond campus that seek to understand our past, present and future, our values and identities and the essential issues we face as individuals and communities.</p>
</div></div></div>    </div>
